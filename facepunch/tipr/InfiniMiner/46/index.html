[/QUOTE] InfiniminerClient > Engines > InterfaceEngine.cs Go there, then find the function Render(), and scroll down to spriteBatch.End(). Then, add this one line above it: [code]DrawTextOutline(spriteBatch, drawRect.Width - 100, drawRect.Height - 305, "Your text here");[/code] Then make a new function: [code] private void DrawTextOutline(SpriteBatch spr, float X, float Y, string str) { Vector2 SZ = uiFont.MeasureString(str); float Wi = SZ.X / 2; float He = SZ.Y / 2; X -= Wi; Y -= He; X = (float)Math.Round(X); Y = (float)Math.Round(Y); Color DrawCol = new Color(Color.YellowGreen, 127); Color OutCol = new Color(Color.Black, 127); spr.DrawString(uiFont, str, new Vector2(X, Y - 1), OutCol); spr.DrawString(uiFont, str, new Vector2(X + 1, Y), OutCol); spr.DrawString(uiFont, str, new Vector2(X, Y + 1), OutCol); spr.DrawString(uiFont, str, new Vector2(X - 1, Y), OutCol); spr.DrawString(uiFont, str, new Vector2(X, Y), DrawCol); }[/code]
